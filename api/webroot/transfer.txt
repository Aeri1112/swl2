	function fetchAllAuctionsBySeller($username)
	{
		$userid = getuserid($username);
		$stmt = $this->db->prepare("SELECT * FROM jedi_city_ah_items WHERE seller = :userid ORDER BY endtime ASC");
		$stmt->execute(['userid' => $userid]);
		$auctions = $stmt->fetchAll(PDO::FETCH_CLASS, "Auktionshaus");
		return $auctions;		
	}
	
	function fetchAllAuctionByBidder($userid)
	{
		$stmt = $this->db->prepare("SELECT * FROM `jedi_city_ah_items` WHERE auctionid = ANY (SELECT auctionid FROM `jedi_city_ah_offers` WHERE userid = :userid GROUP BY auctionid)");
		$stmt->execute(['userid' => $userid]);
		$auctions = $stmt->fetchAll(PDO::FETCH_CLASS, "Auktionshaus");
		return $auctions;
	}
